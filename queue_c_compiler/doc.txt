queue - компилируемый статически типизированый функциональный язык программирования

лексемы языка:

var - говорит компилятору, что создаваемый объект, это переменная
arr - говорит компилятору, что создаваемый объект, это массив
fun - говорит компилятору, что создаваемый объект, это функция
struct - говорит компилятору, что создаваемый объект, это структура

ret - возвращает какой-то результат
out - вывод данных на консоль
in - ввод данных с консоли
if - ключевое слово условий
else if - ключевое слово условий, выполняющееся если if - ложь
else - ключевое слово "иначе"
for - цикл
while - цикл с предусловием
do - ключевое слово для создания цикла do-while

short - целочисленный тип. 2 байта
int - целочисленный тип. 4 байта
long - целочисленный тип. 8 байт
char - тип символа. 1 байт
flt - тип числа с плавающей точкой. 32 байта
bool - логический тип. 1 байт
unsign - объект становится беззнакового типа
pointer - объект - это указатель

@... - разыменование указателя
&... - получение адреса объекта

true - значение типа bool. истина
false - значение типа bool. ложь

? - нулевое значение

= - присвоение
+= - прибавить к правому операнду левый операнд
-= - убавить от правого операнда левый операнд
*= - умножить правый операнд на левый операнд
/= - поделить правый операнд на левый операнд
%= - присвоить остаток от деления правого операнда на левый операнд, к правому операнду
; - окончание строки
. - точка
, - запятая
'' - одинарные ковычки
:: - обращение к элементам структур
-> - указатель инструкций

/- ... -/ - комментарий

+ - знак плюса
- - знак минуса
* - знак умножения
/ - знак деления
% - знак остатка от деления

<= - меньше либо равно
< - меньше
== - равно
=> - больше либо равно
> - больше
!= - не равно

&& - и
|| - или

--*...* - префиксный декремент
*...*-- - постфиксный декремент
++*...* - префиксный инкремент
*...*++ - постфиксный инкремент


спецификаторы для ввода/вывода:

%sdi - для вывода чисел типа int
%sdf - для вывода чисел типа flt
%pch - для вывода символов
%pbl - для вывода булевого типа(выводит true/false), для булевых типов можно так же использовать спецификатор %di(выведет 1/0)

%udi - для вывода чисел типа unsign int
